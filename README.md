# Команды (Извините что на английском)
  cls       - Clear screen  
  dir       - List files  
  cd        - Change directory  
  help      - Show this help  
  exit      - Shutdown system  
  brows     - Text browser (brows http://example.com)  
  playmp3   - Play mp3 file  
  stop      - Stop MP3 playback  
  echo      - Print text to screen  
  ver       - Show version info  
  sysinfo   - Show system information  
  tree      - Display directory structure  
  copy      - Copy file  
  delete    - Delete file  
  edit      - Edit file with Notepad  
  mkdir     - Create directory  
  mklink    - Create symbolic link (not supported)  
  rmdir     - Remove directory  
  color     - Change text/background color  
  time      - Show current time  
  date      - Show current date  
  reboot    - Simulate system reboot  
  calc      - Simple calculator
